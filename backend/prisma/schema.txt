model User {
  id       String  @id @default(uuid())
  email    String  @unique
  password String // hashed with bcrypt
  name     String
  avatar   String? // URL to avatar image

  // Timestamps
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // Relations
  transactions   Transaction[]
  categories     Category[]
  budgetGoals    BudgetGoal[]
  notifications  Notification[]
  userStats      UserStats?
  activityLogs   ActivityLog[]
  userBadges     UserBadge[]
  userChallenges UserChallenge[]
  exportLogs     ExportLog[]
  forecasts      Forecast[]

  @@map("users")
}

model UserStats {
  id     String @id @default(uuid())
  userId String @unique @map("user_id")

  // XP and Level
  xp    Int @default(0)

  // Streak tracking
  streak            Int       @default(0)
  longestStreak     Int       @default(0) @map("longest_streak")
  lastTransactionAt DateTime? @map("last_transaction_at")

  // Stats

  totalIncome       Int @default(0) @map("total_income")
  totalExpense      Int @default(0) @map("total_expense")

  // Gamification flags
  hasExported Boolean @default(false) @map("has_exported")
  hasAnalyzed Boolean @default(false) @map("has_analyzed")

  // Timestamps
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // Relations
  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@map("user_stats")
}

model Challenge {
  id          String        @id @default(uuid())
  title       String
  description String
  target      Int // Target number to achieve
  xpReward    Int           @default(50) @map("xp_reward")
  type        ChallengeType @default(WEEKLY_TRANSACTIONS)

  // Timestamps
  createdAt DateTime @default(now()) @map("created_at")

  // Relations
  userChallenges UserChallenge[]

  @@map("challenges")
}

model UserChallenge {
  id          String    @id @default(uuid())
  userId      String    @map("user_id")
  challengeId String    @map("challenge_id")
  current     Int       @default(0) // Current progress
  isCompleted Boolean   @default(false) @map("is_completed")
  completedAt DateTime? @map("completed_at")
  deadline    DateTime // Challenge deadline (usually end of week)
  weekNumber  Int       @map("week_number") // ISO week number
  year        Int // Year for tracking

  // Timestamps
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // Relations
  user      User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  challenge Challenge @relation(fields: [challengeId], references: [id], onDelete: Cascade)

  @@unique([userId, challengeId, weekNumber, year])
  @@map("user_challenges")
}

enum ChallengeType {
  WEEKLY_TRANSACTIONS
  MONTHLY_SAVINGS
  BUDGET_GOALS
  ANALYSIS_COUNT
  STREAK_MAINTAIN
}

model Transaction {
  id          String          @id @default(uuid())
  userId      String          @map("user_id")
  description String
  amount      Int // Amount in IDR (smallest unit)
  type        TransactionType
  date        DateTime        @db.Date

  // Optional category
  categoryId String? @map("category_id")

  // Clustering
  clusterId String? @map("cluster_id")

  // Source tracking
  source      TransactionSource @default(MANUAL) @map("source")
  csvImportId String?           @map("csv_import_id")

  // Timestamps
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // Relations
  user     User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  category Category? @relation(fields: [categoryId], references: [id])
  cluster  Cluster?  @relation(fields: [clusterId], references: [id])

  @@index([userId, date])
  @@index([userId, type])
  @@index([userId, categoryId])
  @@map("transactions")
}

enum TransactionType {
  INCOME
  EXPENSE
}

enum TransactionSource {
  MANUAL
  CSV_IMPORT
  API
}

model Category {
  id              String  @id @default(uuid())
  userId          String  @map("user_id")
  name            String
  color           String  @default("#888888")
  icon            String? // Icon name
  isAutoGenerated Boolean @default(false) @map("is_auto_generated")

  // AI-generated metadata
  aiConfidence Float?   @map("ai_confidence") // Confidence score from clustering
  aiKeywords   String[] @map("ai_keywords") // Keywords extracted from transactions

  // Timestamps
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // Relations
  user         User          @relation(fields: [userId], references: [id], onDelete: Cascade)
  transactions Transaction[]
  budgetGoals  BudgetGoal[]
  forecasts    Forecast[]

  @@unique([userId, name])
  @@map("categories")
}

model BudgetGoal {

  alertSent100   Boolean @default(false) @map("alert_sent_100")

  // Timestamps

  updatedAt DateTime @updatedAt @map("updated_at")

  // Relations

  category Category @relation(fields: [categoryId], references: [id], onDelete: Cascade)

  @@unique([userId, categoryId])
  @@map("budget_goals")
}

enum BudgetPeriod {
  WEEKLY
  MONTHLY
  YEARLY
}

model Cluster {

  @@map("clusters")

model Notification {
  id      String           @id @default(uuid())
  userId  String           @map("user_id")
  title   String
  message String
  type    NotificationType
  isRead  Boolean          @default(false) @map("is_read")

  // Additional data (JSON for flexibility)
  metadata Json? // e.g., { transactionId, categoryId, badgeId, etc. }

  // Timestamps
  createdAt DateTime  @default(now()) @map("created_at")
  readAt    DateTime? @map("read_at")

  // Relations
  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId, isRead])
  @@index([userId, createdAt])
  @@map("notifications")
}

enum NotificationType {
  BUDGET_ALERT // Budget threshold reached
  PATTERN_FOUND // New cluster/pattern discovered
  ACHIEVEMENT // Badge unlocked
  LEVEL_UP // Level increased
  REMINDER // Daily reminder
  STREAK_WARNING // Streak about to break
  CHALLENGE_COMPLETE // Weekly challenge completed
}

  @@map("forecast_cache")

model Forecast {

  @@unique([userId, categoryId, targetMonth, targetYear])
  @@map("forecasts")
